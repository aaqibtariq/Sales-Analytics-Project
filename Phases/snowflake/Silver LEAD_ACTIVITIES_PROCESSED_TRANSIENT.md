
# Create table 

```

CREATE OR REPLACE TABLE SILVER.LEAD_ACTIVITIES_PROCESSED_TRANSIENT AS
SELECT
    activity.value AS JSON_OBJECT,
    r.JSON_OBJECT:insert_date::TIMESTAMP_NTZ AS INSERT_DATE
FROM BRONZE.LEAD_ACTIVITIES_RAW r,
LATERAL FLATTEN(INPUT => r.JSON_OBJECT:raw_data:data) activity;

```

# Validate 

```

SELECT COUNT(*) FROM SILVER.LEAD_ACTIVITIES_PROCESSED_TRANSIENT;
```
