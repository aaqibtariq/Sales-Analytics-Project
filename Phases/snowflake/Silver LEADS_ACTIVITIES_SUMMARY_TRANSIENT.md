# create table

```
CREATE OR REPLACE TABLE SILVER.LEADS_ACTIVITIES_SUMMARY_TRANSIENT AS
SELECT
    ACTIVITY_ID,
    LEAD_ID,
    CONTACT_ID,
    USER_ID,
    TYPE,
    TEXT,
    STATUS_CHANGE,
    CREATED_BY_NAME,
    UPDATED_BY_NAME,
    ATTENDEES,
    CUSTOM_ACTIVITY_ID,
    MEETING_STARTS_AT,
    MEETING_END_AT,
    MEETING_STATUS,
    ACTIVITY_AT,
    DATE_CREATED,
    DATE_UPDATED,
    CREATED_BY_USER,
    UPDATED_BY_USER,
    INSERT_DATE,
    UPDATE_DATE,
    MD5_HASH
FROM SILVER.LEAD_ACTIVITIES_PROCESSED;


```
