# Changes in lead activities summary

```

Before we create ALL_STRATEGIES_DETAILS, let's make one final enhancement to LEADS_ACTIVITIES_SUMMARY:
Add these business columns:
SETTER_USER_ID
SETTER_EMAIL
SETTER_NAME
CLOSER_USER_ID
CLOSER_EMAIL
CLOSER_NAME
OFFER_PRESENTED
PROGRAM
CONTRACT_VALUE
CASH_COLLECTED
OBJECTIONS_FACED
This is not changing the SME architecture. It's enriching the existing Silver summary with business attributes we already
proved exist in Bronze. The Gold views then become simple reporting views built from Silver, which is exactly how a Medallion
 architecture is intended to work.
The remaining four Gold views (ALL_STRATEGIES_DETAILS, SALES_DETAILS, OUTBOUND_STRATEGIES_BOOKED,
 and OUTBOUND_PROSPECT_DIALS) will all reuse these same business fields. It avoids repeatedly reaching back into Bronze and keeps
Gold clean and maintainable.


USE DATABASE SALES_ANALYTICS_DB;
USE SCHEMA SILVER;

ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS SETTER_USER_ID STRING;
ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS SETTER_EMAIL STRING;
ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS SETTER_NAME STRING;

ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS CLOSER_USER_ID STRING;
ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS CLOSER_EMAIL STRING;
ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS CLOSER_NAME STRING;

ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS OFFER_PRESENTED STRING;
ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS PROGRAM STRING;
ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS CONTRACT_VALUE STRING;
ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS CASH_COLLECTED STRING;
ALTER TABLE LEADS_ACTIVITIES_SUMMARY ADD COLUMN IF NOT EXISTS OBJECTIONS_FACED STRING;



TRUNCATE TABLE SILVER.LEADS_ACTIVITIES_SUMMARY;

INSERT INTO SILVER.LEADS_ACTIVITIES_SUMMARY
WITH CAD_DEDUP AS (
    SELECT *
    FROM SILVER.CUSTOM_ACTIVITIES_ALL_LEADS_DETAILS
    QUALIFY ROW_NUMBER() OVER (
        PARTITION BY ACTIVITY_ID
        ORDER BY UPDATE_DATE DESC, INSERT_DATE DESC
    ) = 1
),

BRONZE_CUSTOM_FIELDS AS (
    SELECT
        activity.value:id::STRING AS ACTIVITY_ID,
        activity.value:lead_id::STRING AS LEAD_ID,

        activity.value:"custom.cf_h3tYb9J6yPK7J4PMExDGsEqPCf8kBGBrRNIur2Dm5aN"::STRING AS TRIAGE_CALL_OUTCOME,
        activity.value:"custom.cf_dhJR4N7Rm6czuJthYGJP6KqUcuOzi7fqApGI7puWnMo"::STRING AS STRATEGY_CALL_OUTCOME,

        activity.value:"custom.cf_LGyzSTMPy37y87rDOUmFXlZA42HPSIpbipeT2OsQNHW"::STRING AS OFFER_PRESENTED,
        activity.value:"custom.cf_mXqKxcmjnlEW223wM4lgb04XyYwDm0GJ9v99xWHPWO2"::STRING AS PROGRAM,
        activity.value:"custom.cf_vIanPjPEit6ssajmWkcprF2V1nO1itfes8hOSnjmhfT"::STRING AS CONTRACT_VALUE,
        activity.value:"custom.cf_eyLbGJm9DYY7cuJk2otnCxhUEzK9ayEARiE81xPG5uY"::STRING AS CASH_COLLECTED,
        activity.value:"custom.cf_aIN5Gtqq33tUCCBxFTW63FY6d3mofnKIfFqfWPkvNla"::STRING AS OBJECTIONS_FACED,

        activity.value:"custom.cf_v385AJ8HSgepKQ3rvqo4yOA3nn49eGqz39DOqojJG5M"::STRING AS SETTER_USER_ID,
        activity.value:"custom.cf_Lv5lSqLOZwLrNhe5M7kWx2mF8Ge2Z23aw5NUNhbXvVS"::STRING AS CLOSER_USER_ID

    FROM SALES_ANALYTICS_DB.BRONZE.LEAD_ACTIVITIES_RAW r,
    LATERAL FLATTEN(INPUT => r.JSON_OBJECT:raw_data:data) activity

    QUALIFY ROW_NUMBER() OVER (
        PARTITION BY activity.value:lead_id::STRING, activity.value:id::STRING
        ORDER BY activity.value:activity_at::TIMESTAMP_NTZ DESC,
                 activity.value:date_updated::TIMESTAMP_NTZ DESC
    ) = 1
)

SELECT
    lap.ACTIVITY_ID,
    lap.LEAD_ID,
    lap.CONTACT_ID,
    lap.USER_ID,
    lap.TYPE,
    lap.TEXT,
    lap.STATUS_CHANGE,
    lap.CREATED_BY_NAME,
    lap.UPDATED_BY_NAME,
    lap.ATTENDEES,
    lap.CUSTOM_ACTIVITY_ID,
    lap.MEETING_STARTS_AT,
    lap.MEETING_END_AT,
    lap.MEETING_STATUS,
    lap.ACTIVITY_AT,
    lap.DATE_CREATED,
    lap.DATE_UPDATED,
    lap.CREATED_BY_USER,
    lap.UPDATED_BY_USER,
    CURRENT_TIMESTAMP(),
    CURRENT_TIMESTAMP(),
    cad.WRAPPED_ACTIVITIES,
    ca.CUSTOM_ACTIVITY_NAME AS CUSTOM_ACTIVITY,

    CASE
        WHEN ca.CUSTOM_ACTIVITY_NAME = '3) Triage Call' THEN 'Triage Call Outcome'
        WHEN ca.CUSTOM_ACTIVITY_NAME = '5) Strategy Call' THEN 'Strategy Call Outcome'
        ELSE ca.CUSTOM_ACTIVITY_OUTCOME_NAME
    END AS CUSTOM_ACTIVITY_OUTCOME_NAME,

    CASE
        WHEN ca.CUSTOM_ACTIVITY_NAME = '3) Triage Call' THEN bcf.TRIAGE_CALL_OUTCOME
        WHEN ca.CUSTOM_ACTIVITY_NAME = '5) Strategy Call' THEN bcf.STRATEGY_CALL_OUTCOME
        ELSE ca.CUSTOM_ACTIVITY_OUTCOME_ID
    END AS CUSTOM_ACTIVITY_OUTCOME,

    cu.FIRST_NAME || ' ' || cu.LAST_NAME AS DEA_INTERNAL_NAME,
    cu.EMAIL AS DEA_INTERNAL_EMAIL,

    bcf.SETTER_USER_ID,
    su.EMAIL AS SETTER_EMAIL,
    su.FIRST_NAME || ' ' || su.LAST_NAME AS SETTER_NAME,

    bcf.CLOSER_USER_ID,
    clu.EMAIL AS CLOSER_EMAIL,
    clu.FIRST_NAME || ' ' || clu.LAST_NAME AS CLOSER_NAME,

    bcf.OFFER_PRESENTED,
    bcf.PROGRAM,
    bcf.CONTRACT_VALUE,
    bcf.CASH_COLLECTED,
    bcf.OBJECTIONS_FACED

FROM SILVER.LEAD_ACTIVITIES_PROCESSED lap
LEFT JOIN CAD_DEDUP cad
    ON lap.ACTIVITY_ID = cad.ACTIVITY_ID
LEFT JOIN BRONZE_CUSTOM_FIELDS bcf
    ON lap.ACTIVITY_ID = bcf.ACTIVITY_ID
   AND lap.LEAD_ID = bcf.LEAD_ID
LEFT JOIN SILVER.CUSTOM_ACTIVITIES ca
    ON lap.CUSTOM_ACTIVITY_ID = ca.CUSTOM_ACTIVITY_ID
LEFT JOIN SILVER.CLOSE_CRM_USERS_PROCESSED cu
    ON lap.USER_ID = cu.USER_ID
LEFT JOIN SILVER.CLOSE_CRM_USERS_PROCESSED su
    ON bcf.SETTER_USER_ID = su.USER_ID
LEFT JOIN SILVER.CLOSE_CRM_USERS_PROCESSED clu
    ON bcf.CLOSER_USER_ID = clu.USER_ID;

SELECT
```

# Create gold layer

```
USE DATABASE SALES_ANALYTICS_DB;
USE SCHEMA GOLD;

CREATE OR REPLACE VIEW GOLD.ALL_STRATEGIES_DETAILS AS
SELECT
    LEAD_ID,

    ACTIVITY_AT::DATE AS ACTIVITY_LOG_DATE,

    CUSTOM_ACTIVITY,

    -- CRM record status, such as published or draft
    STATUS_CHANGE AS STATUS,

    CLOSER_EMAIL,
    CLOSER_NAME,

    -- Populated for Strategy Calls;
    -- naturally NULL for Strategy Call Follow Ups
    CUSTOM_ACTIVITY_OUTCOME AS STRATEGY_CALL_OUTCOME,

    ACTIVITY_AT::DATE AS DATE_OF_STRATEGY_CALL,

    OFFER_PRESENTED,

    -- SME model uses SETTER plus SETTER_NAME.
    -- SETTER is represented by the setter's email.
    SETTER_EMAIL AS SETTER,
    SETTER_NAME

FROM SALES_ANALYTICS_DB.SILVER.LEADS_ACTIVITIES_SUMMARY
WHERE CUSTOM_ACTIVITY IN (
    '5) Strategy Call',
    '6) Strategy Call Follow Up'
);

```
