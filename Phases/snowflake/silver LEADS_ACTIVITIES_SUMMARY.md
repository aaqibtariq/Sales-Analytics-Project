# create table 

```
CREATE OR REPLACE TABLE SILVER.LEADS_ACTIVITIES_SUMMARY (
    ACTIVITY_ID STRING,
    LEAD_ID STRING,
    CONTACT_ID STRING,
    USER_ID STRING,
    TYPE STRING,
    TEXT STRING,
    STATUS_CHANGE STRING,
    CREATED_BY_NAME STRING,
    UPDATED_BY_NAME STRING,
    ATTENDEES VARIANT,
    CUSTOM_ACTIVITY_ID STRING,
    MEETING_STARTS_AT TIMESTAMP_NTZ,
    MEETING_END_AT TIMESTAMP_NTZ,
    MEETING_STATUS STRING,
    ACTIVITY_AT TIMESTAMP_NTZ,
    DATE_CREATED TIMESTAMP_NTZ,
    DATE_UPDATED TIMESTAMP_NTZ,
    CREATED_BY_USER STRING,
    UPDATED_BY_USER STRING,
    INSERT_DATE TIMESTAMP_NTZ,
    UPDATE_DATE TIMESTAMP_NTZ,
    WRAPPED_ACTIVITIES VARIANT,
    CUSTOM_ACTIVITY STRING,
    CUSTOM_ACTIVITY_OUTCOME_NAME STRING,
    CUSTOM_ACTIVITY_OUTCOME STRING,
    DEA_INTERNAL_NAME STRING,
    DEA_INTERNAL_EMAIL STRING
);

```

# insert

```

INSERT INTO SILVER.LEADS_ACTIVITIES_SUMMARY
WITH CAD_DEDUP AS (
    SELECT *
    FROM SILVER.CUSTOM_ACTIVITIES_ALL_LEADS_DETAILS
    QUALIFY ROW_NUMBER() OVER (
        PARTITION BY ACTIVITY_ID
        ORDER BY UPDATE_DATE DESC, INSERT_DATE DESC
    ) = 1
)
SELECT
    lap.ACTIVITY_ID,
    lap.LEAD_ID,
    lap.CONTACT_ID,
    lap.USER_ID,
    lap.TYPE,
    lap.TEXT,
    lap.STATUS_CHANGE,
    lap.CREATED_BY_NAME,
    lap.UPDATED_BY_NAME,
    lap.ATTENDEES,
    lap.CUSTOM_ACTIVITY_ID,
    lap.MEETING_STARTS_AT,
    lap.MEETING_END_AT,
    lap.MEETING_STATUS,
    lap.ACTIVITY_AT,
    lap.DATE_CREATED,
    lap.DATE_UPDATED,
    lap.CREATED_BY_USER,
    lap.UPDATED_BY_USER,
    CURRENT_TIMESTAMP(),
    CURRENT_TIMESTAMP(),
    cad.WRAPPED_ACTIVITIES,
    ca.CUSTOM_ACTIVITY_NAME AS CUSTOM_ACTIVITY,
    ca.CUSTOM_ACTIVITY_OUTCOME_NAME,
    ca.CUSTOM_ACTIVITY_OUTCOME_ID AS CUSTOM_ACTIVITY_OUTCOME,
    cu.FIRST_NAME || ' ' || cu.LAST_NAME AS DEA_INTERNAL_NAME,
    cu.EMAIL AS DEA_INTERNAL_EMAIL
FROM SILVER.LEAD_ACTIVITIES_PROCESSED lap
LEFT JOIN CAD_DEDUP cad
    ON lap.ACTIVITY_ID = cad.ACTIVITY_ID
LEFT JOIN SILVER.CUSTOM_ACTIVITIES ca
    ON lap.CUSTOM_ACTIVITY_ID = ca.CUSTOM_ACTIVITY_ID
LEFT JOIN SILVER.CLOSE_CRM_USERS_PROCESSED cu
    ON lap.USER_ID = cu.USER_ID;


```

# extra validation


```
-- 1. Base table count
SELECT COUNT(*)
FROM SILVER.LEAD_ACTIVITIES_PROCESSED;

--2. Distinct activity IDs

SELECT
    COUNT(*) AS TOTAL_ROWS,
    COUNT(DISTINCT ACTIVITY_ID) AS DISTINCT_ACTIVITY_IDS
FROM SILVER.CUSTOM_ACTIVITIES_ALL_LEADS_DETAILS;


--3. Check duplicate users

SELECT
    USER_ID,
    COUNT(*)
FROM SILVER.CLOSE_CRM_USERS_PROCESSED
GROUP BY USER_ID
HAVING COUNT(*) > 1;


--4. Check duplicate custom activities
SELECT
    CUSTOM_ACTIVITY_ID,
    COUNT(*)
FROM SILVER.CUSTOM_ACTIVITIES
GROUP BY CUSTOM_ACTIVITY_ID
HAVING COUNT(*) > 1;




```
