# Create bronze tables

```

USE DATABASE SALES_ANALYTICS_DB;
USE SCHEMA BRONZE;

CREATE OR REPLACE TABLE CUSTOM_ACTIVITIES_RAW (
    JSON_OBJECT VARIANT,
    INSERT_DATE TIMESTAMP_NTZ
);

CREATE OR REPLACE TABLE LEAD_ACTIVITIES_RAW (
    JSON_OBJECT VARIANT,
    INSERT_DATE TIMESTAMP_NTZ
);

CREATE OR REPLACE TABLE CLOSE_CRM_USERS_RAW (
    JSON_OBJECT VARIANT,
    INSERT_DATE TIMESTAMP_NTZ
);

```

# CUSTOM ACTIVITIES RAW

```
COPY INTO CUSTOM_ACTIVITIES_RAW (JSON_OBJECT, INSERT_DATE)
FROM (
    SELECT
        $1,
        CURRENT_TIMESTAMP()
    FROM @SALES_ANALYTICS_RAW_STAGE/custom_activities_raw/
)
FILE_FORMAT = JSON_FF
ON_ERROR = CONTINUE;

```
