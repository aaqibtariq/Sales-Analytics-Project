# Create DB

```
CREATE DATABASE IF NOT EXISTS SALES_ANALYTICS_DB;
```

# Create Schema 

```
CREATE SCHEMA IF NOT EXISTS SALES_ANALYTICS_DB.BRONZE;
CREATE SCHEMA IF NOT EXISTS SALES_ANALYTICS_DB.SILVER;
CREATE SCHEMA IF NOT EXISTS SALES_ANALYTICS_DB.GOLD;
CREATE SCHEMA IF NOT EXISTS SALES_ANALYTICS_DB.EXPORTS;
```

# create tables

```
CREATE OR REPLACE TABLE SALES_ANALYTICS_DB.BRONZE.CUSTOM_ACTIVITIES_RAW (
    JSON_OBJECT VARIANT,
    INSERT_DATE TIMESTAMP_NTZ
);

CREATE OR REPLACE TABLE SALES_ANALYTICS_DB.BRONZE.LEAD_ACTIVITIES_RAW (
    JSON_OBJECT VARIANT,
    INSERT_DATE TIMESTAMP_NTZ
);

CREATE OR REPLACE TABLE SALES_ANALYTICS_DB.BRONZE.CLOSE_CRM_USERS_RAW (
    JSON_OBJECT VARIANT,
    INSERT_DATE TIMESTAMP_NTZ
);

```

# Create Filesytem

```
CREATE OR REPLACE FILE FORMAT JSON_FF
TYPE = JSON
STRIP_OUTER_ARRAY = FALSE;
```

# Create S3 storage integration

```
CREATE OR REPLACE STORAGE INTEGRATION SALES_ANALYTICS_S3_INT
TYPE = EXTERNAL_STAGE
STORAGE_PROVIDER = S3
ENABLED = TRUE
STORAGE_AWS_ROLE_ARN = 'arn:aws:iam:: paste your role'
STORAGE_ALLOWED_LOCATIONS = ('s3://paste your path');

# Get Snowflake IAM values
DESC INTEGRATION SALES_ANALYTICS_S3_INT;

# Copy these two vallues and paste into your AWS IAM role trust policy

STORAGE_AWS_IAM_USER_ARN
STORAGE_AWS_EXTERNAL_ID
```
